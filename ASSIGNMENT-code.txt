    <script>
        function signIn() {
            var username = document.getElementById('username').value;
            var password = document.getElementById('password').value;

            // Validate login credentials
            if (validateLogin(username, password)) {
                if (username === 'admin') {
                    showAdminForm();
                } else if (username === 'customer1' || username === 'customer2') {
                    showCustomerForm(username);
                }
            } else {
                alert('Invalid username or password');
            }
        }

        function validateLogin(username, password) {
            var dummyUsernames = ['admin', 'customer1', 'customer2'];
            var dummyPasswords = ['admin', 'customer1', 'customer2'];

            for (var i = 0; i < dummyUsernames.length; i++) {
                if (username === dummyUsernames[i]) {
                    return password === dummyPasswords[i];
                }
            }

            return false;
        }

        function showAdminForm() {
            alert('Welcome, admin!');
        }

        function showCustomerForm(username) {
            var customerForm = document.createElement('div');
            customerForm.className = 'container';

            var titleLabel = document.createElement('h2');
            titleLabel.textContent = 'Data Entry for ' + username;
            customerForm.appendChild(titleLabel);

            var dataField = document.createElement('input');
            dataField.type = 'text';
            dataField.placeholder = 'Enter data';
            customerForm.appendChild(dataField);

            var submitButton = document.createElement('button');
            submitButton.textContent = 'Submit';
            submitButton.addEventListener('click', function () {
                var data = dataField.value;

                // ...

                alert('Data submitted successfully!');
                customerForm.remove(); // Close the customer data entry form
            });
            customerForm.appendChild(submitButton);

            document.body.appendChild(customerForm);
        }
